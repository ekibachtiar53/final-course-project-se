Pseudocode Peminjaman dan Pengembalian Buku

READ anggota 
IF bukan anggota
	INPUT pendaftaran 
ELSE   
	INPUT Pengisian daftar hadir

IF peminjam buku
	INPUT pencarian buku
IF ELSE pengembalian buku
	READ pengembalian buku

IF pegengembalian buku
	READ proese pengembalian 
	IF ELSE pembayaran terlambat 
		WRITE pembayaran denda
	ELSE 
		pencatatan dokumen
		IF Peminjaman buku
			INPUT pencarian buku
		ELSE 
			selesai

INPUT pencarian buku
OUTPUT buku
READ pemilihan buku

IF tambahan buku
	INPUT pencarian buku
ELSE 
	Selesai
